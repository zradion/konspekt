cmd команды в Windows

оболочка - программа с которой вы взаимодействуете при выполнении команд в терминале (PowerShell).
Интерфейс командной строки(CLI) - программа, предназначенная для запуска в оболочке внутри терминала.
cwd - current working directory
stdout - стандарный поток вывода
stderr - стандартный поток ошибок

cmdlets - команды в PowerShell
pwd - показать путь к текущему каталоку(print working directory)
mkdir FOLDERPATH - создаь новый каталог
ni FILEPATH - создать новый файл
clear - очистить окно терминала
ls - список содержимого папки (list); (dir)
ls -al - список всего содержимого папки с информацией(два флага all; long)
cat TARGET - показать содержание TARGET (concatenete)
cd FOLDERPATH - перейти в каталог (change directory)
cd .. - перейти в родительский каталог
echo TEXT - показать текст в терминале
echo TEXT > TARGET - печать TEXT в файл с именем TARGET
echo TEXT >> TARGET - добавить TEXT в TARGET
cp SOURCE TARGET - копировать SOURCE в TARGET
rni SOURCE TARGET - переименовать SOURCE в TARGET
python PYTHONFILE - запустить PYTHONFILE

echo отправляет строку в stdout; echo > name.py - в файл (создаст или перезапишет); cat name.py - выполнить файл или несколько файлов один за другим; Переменные среды - переменные с инфо о состоянии среды в которой работает терминал.
PATH - переменная среды со списком каталогов.

python - без аргументов - запустить интерактивный интерпритатор Python(REPL)
cat ENV:Path - вывести какие пути в вашем PATH
pip list - список пакетов глобального окружения установленных на компе

Get-ExecutionPolicy - проверить разрешены ли сценарии выполнения
Set-ExecutionPolicy RemoteSigned - запустить в PowerShell от админа, чтобы установить виртуальную среду; выбрать А (для всех) (unrestricted)
Set-ExecutionPolicy Default - вернуть запрет на использование виртуальной среды
python -m venv venv - создать папку виртуального окружения
/venv/Scripts/activate - активировать
deactivate - деактивировать
