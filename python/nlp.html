natasha

from natasha import (
Segmenter, MorphVocab, NewsEmbedding, NewsMorphTagger,
NewsSyntaxParser, NewsNERTagger, PER, NamesExtractor, Doc)

segmenter = Segmenter()
morph_vocab = MorphVocab()
emb = NewsEmbedding()
morph_tagger = NewsMorphTagger(emb)
syntax_parser = NewsSyntaxParser(emb)
ner_tagger = NewsNERTagger(emb)
names_extractor = NamesExtractor(morph_vocab)
doc = Doc(text)

Segmentation
doc.segment(segmenter)
doc.tokens[:5] - 5 первых токенов (start, stop, text)
doc.sents[:5] - 5 первых предложений (start, stop, text)

Morphology
doc.tag_morph(morph_tagger)
doc.tokens[:5] - 5 первых токенов (start, stop, text, pos, feats)
pos='noun, propn, adp, verb, punct, sconj, adj, det, pron, part, adv, cconj'
feats=dict(Animacy='Anim, Inan', Case='Nom, Gen, Loc', Gender='Masc, Fem, Neut', Number='Sing, Plur', Aspect='Perf', Mood='Ind', Tense='Past', VerbForm='Fin, Conv, Part', Voice='Act, Pass')

Lemmatization
for token in doc.tokens:
    token.lemmatize(morph_vocab)
doc.tokens[:5] - 5 первых токенов (start, stop, text, pos, feats, lemma)
